# Mining_Signals tool runtime dependencies.
# Anything imported by ``ocr/sc_ocr/*``, ``ocr/signal_anchor.py``, the

requests>=2.28.0

#                              all per-glyph crop math)

#                              CC labelling for icon validation)

#   * ocr/refinery_reader.py + others

# ONNX (model file format) — needed by training/export scripts that
